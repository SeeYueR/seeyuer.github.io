function addLanguage() {
  // 获取所有 .code-wrapper 元素
  var wrappers = $(".code-wrapper"); // 此处改动
  for (var i = 0; i < wrappers.length; i++) {
    var wrapper = wrappers[i];
    // 获取代码语言
    var lang = wrapper.firstChild.firstChild.classList[1]; // 此处改动

    // 折叠块的 id
    var id = `kiyan-collapse-${getUuiD()}`;
    // 前面折叠按钮
    var btn = `<i class="fas fa-angle-down rotate" type="button" data-toggle="collapse" data-target="#${id}"></i>`;
    // 代码语言
    var span = `<span>${getLanguage(lang)}</span>`;

    // 如果代码行数超过30行，设置折叠块的默认状态为折叠
    var isLargeCode = wrapper.innerText.split('\n').length > 30;
    var defaultCollapseClass = isLargeCode ? 'collapse' : 'collapse show';

    // 折叠块包裹原来的内容
    var div = `<div class="${defaultCollapseClass}" id="${id}">${wrapper.innerHTML}</div>`;
    wrapper.innerHTML = btn + span + div;

    // 如果代码行数超过30行，为折叠按钮添加点击事件
    if (isLargeCode) {
      var collapseButton = wrapper.querySelector(".fa-angle-down");
      collapseButton.addEventListener('click', function () {
        var icon = $(this);
        icon.removeClass('rotate');
      });
    }
  }
}

function copyText(text) {
  navigator.clipboard.writeText(text);
}

// 添加复制按钮
function addCopyButton(wrapper) {

  var copyBtn = document.createElement('i');
  copyBtn.className = 'far fa-copy copy-code copy-right';
  
  copyBtn.addEventListener('click', function() {
    copyText(wrapper.innerText);  
    copyBtn.innerHTML = '<i class="fas fa-check"></i>';
    setTimeout(function() {
      copyBtn.innerHTML = '';
    }, 1000); 
  });

  wrapper.prepend(copyBtn);
}
